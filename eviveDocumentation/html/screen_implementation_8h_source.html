847 static void evive_oscilloscope_loop() {
848  //int value = ade791x_read_v1 ();
849  // value = ade791x_read_vim ();
850  // ade791x_read_i();
851  //analogWrite(5,analogRead(A9)/4);
852  if (trig_mode != TRIG_SCAN) {
853  unsigned long st = millis();
854  int oad;
855  if (trig_ch == 0) oad = ConvertMilliVoltToPixel(ade791x_read_v1(), range0, ch0_off);
856  // oad = ade791x_read_v1();
857  else{
858  if (range1 < 9) oad = ConvertMilliVoltToPixel(ade791x_read_vim(), range1, ch1_off);
859  else oad = ConvertMilliVoltToPixel(ade791x_read_im(), range1, ch1_off);
860  }
861  //Wait till trigger detected
862  for (;;) {
863  int ad;
864  if (trig_ch == 0)
865  ad = ConvertMilliVoltToPixel(ade791x_read_v1(), range0, ch0_off);
866  else{
867  if (range1 < 9) ad = ConvertMilliVoltToPixel(ade791x_read_vim(), range1, ch1_off);
868  else ad = ConvertMilliVoltToPixel(ade791x_read_im(), range1, ch1_off);
869  }
870  //Detect trigger
871  if (trig_edge == TRIG_E_UP) {
872  if (ad >= trig_lv && ad > oad+4)
873  break;
874  } else {
875  if (ad <= trig_lv && ad < oad-4)
876  break;
877  }
878  oad = ad;
879 
880  CheckSW();
881  if (trig_mode == TRIG_SCAN)
882  break;
883  if (trig_mode == TRIG_AUTO && (millis() - st) > 100)
884  break;
885  }
886  }
887 
888  // sample and draw depending on the sampling rate
889  if (rate <= 5 && Start) {
890  // change the index for the double buffer
891  if (sample == 0)
892  sample = 2;
893  else
894  sample = 0;
895 
896  if (rate == 0) { // full speed, channel 0 only
897  unsigned long st = millis();
898  for (int i=0; i<SAMPLES; i ++) {
899  data[sample+0][i] = ConvertMilliVoltToPixel(ade791x_read_v1(), range0, ch0_off);;
900  }
901  for (int i=0; i<SAMPLES; i ++)
902  data[sample+1][i] = 0;
903 // Serial.println(millis()-st);
904  } else if (rate == 1) { // full speed, channel 1 only
905  unsigned long st = millis();
906  for (int i=0; i<SAMPLES; i ++) {
907  if (range1 < 9) data[sample+1][i] = ConvertMilliVoltToPixel(ade791x_read_vim(), range1, ch1_off);
908  else data[sample+1][i] = ConvertMilliVoltToPixel(ade791x_read_im(), range1, ch1_off);
909  }
910  for (int i=0; i<SAMPLES; i ++)
911  data[sample+0][i] = 0;
912 // Serial.println(millis()-st);
913  } else if (rate == 2) { // full speed, dual channel
914  unsigned long st = millis();
915  for (int i=0; i<SAMPLES; i ++) {
916  data[sample+0][i] = ConvertMilliVoltToPixel(ade791x_read_v1(), range0, ch0_off);;
917  if (range1 < 9) data[sample+1][i] = ConvertMilliVoltToPixel(ade791x_read_vim(), range1, ch1_off);
918  else data[sample+1][i] = ConvertMilliVoltToPixel(ade791x_read_im(), range1, ch1_off);
919  }
920  // Serial.println(millis()-st);
921  } else if (rate >= 3 && rate <= 5) { // .5ms, 1ms or 2ms sampling
922  const unsigned long r_[] = {5000/DOTS_DIV, 10000/DOTS_DIV, 20000/DOTS_DIV};
923  unsigned long st0 = millis();
924  unsigned long st = micros();
925  unsigned long r = r_[rate - 3];
926  //DO NOT ADD ANY STATEMENT INSIDE FOR LOOP
927  for (int i=0; i<SAMPLES; i ++) {
928  while((st - micros())<r) ;
929  st += r;
930  data[sample+0][i] = ConvertMilliVoltToPixel(ade791x_read_v1(), range0, ch0_off);;
931  if (range1 < 9) data[sample+1][i] = ConvertMilliVoltToPixel(ade791x_read_vim(), range1, ch1_off);
932  else data[sample+1][i] = ConvertMilliVoltToPixel(ade791x_read_im(), range1, ch1_off);
933  }
934  // Serial.println(millis()-st0);
935  }
936  ClearAndDrawGraph();
937  CheckSW();
938  DrawGrid();
939  //DrawText();
940  } else if (Start) { // 5ms - 500ms sampling
941  // copy currently showing data to another
942  if (sample == 0) {
943  for (int i=0; i<SAMPLES; i ++) {
944  data[2][i] = data[0][i];
945  data[3][i] = data[1][i];
946  }
947  } else {
948  for (int i=0; i<SAMPLES; i ++) {
949  data[0][i] = data[2][i];
950  data[1][i] = data[3][i];
951  }
952  }

983  //DrawText();
984  } else {
985  CheckSW();
986  }
